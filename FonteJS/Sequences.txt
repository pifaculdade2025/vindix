CREATE GENERATOR GEN_EMPRESAS_ID;
SET GENERATOR GEN_EMPRESAS_ID TO 0;

-- Trigger
CREATE OR ALTER TRIGGER TRG_EMPRESAS_BI
FOR EMPRESAS
ACTIVE BEFORE INSERT POSITION 0
AS
BEGIN
  IF (NEW.ID IS NULL) THEN
    NEW.ID = GEN_ID(GEN_EMPRESAS_ID, 1);
END
